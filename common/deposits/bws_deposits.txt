d_lost_pops1 = {
	icon = d_metal_boneyard
	time = 500

	is_for_colonizeable = yes
	category = deposit_cat_blockers

	potential = { always = no }

	planet_modifier = {
	}

	on_cleared = {
		if = {
			limit = {
				any_owned_pop = {
					is_gestalt = no
					is_robot_pop = no
				}
			}
			random_owned_pop = {
				limit = {
					is_gestalt = no
					is_robot_pop = no
				}
				root = {
					create_pop = {
						species = prev
					}
					create_pop = {
						species = prev
					}
				}
			}
		}
		else = {
			random_owned_pop = {
				root = {
					create_pop = {
						species = prev
					}
					create_pop = {
						species = prev
					}
				}
			}
		}
	}

	blocker_swap_types = {
		d_lost_pops2
	}

	resources = {
		category = deposit_blockers
		cost = {
			energy = 500
			unity = 200
			influence = 25
		}
	}
}
